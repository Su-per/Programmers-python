```python
def solution(board, moves):
    answer = 0
    basket = []
    for move in list(map(lambda x: x-1, moves)):
        for i in range(0, len(board)):
            if board[i][move] != 0:
                basket.insert(0, board[i][move])
                board[i][move] = 0
                break
        i = 0
        while i < len(basket) - 1:
            if basket[i] == basket[i + 1]:
                del basket[i]
                del basket[i]
                answer, i = answer + 2, i - 1
            i += 1
    return answer
    ```
