```python
import heapq
def solution(n, works):
    heap, answer = [], 0
    for i in works: heapq.heappush(heap, [-i, i])
    for i in range(0,n):
        j = heapq.heappop(heap)
        j[0], j[1] = j[0]+1, j[1]-1
        heapq.heappush(heap, j)
    for i in range(0,len(heap)):
        if heap[i][1] > 0: answer += pow(heap[i][1], 2)
    return answer
```
